// Inside n8n Function Node:

// 1. HTML template stored as string variable
const htmlTemplate = `<!DOCTYPE html><html>...</html>`;

// 2. Replace placeholders with actual data
let finalHTML = htmlTemplate;
finalHTML = finalHTML.replace(/{{customerName}}/g, $json.customerName);
finalHTML = finalHTML.replace(/{{serialNumber}}/g, $json.serialNumber);
finalHTML = finalHTML.replace(/{{warrantyExpiryDate}}/g, $json.warrantyExpiryDate);
finalHTML = finalHTML.replace(/{{daysRemaining}}/g, $json.daysRemaining);

// 3. Return HTML to webhook response
return { json: { html: finalHTML } };
